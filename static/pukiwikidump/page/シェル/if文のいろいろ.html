普通のif文 †

if [ -e $LOCK ]
then
    exit 0
fi
↑

単に1行に書く †

if [ -e $LOCK ]; then exit 0; fi
↑

終了ステータスを利用する †

[ -e $LOCK ] && exit 0
↑

コマンドらしく書く †

test -e $LOCK && exit 0
